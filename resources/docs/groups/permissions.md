```bash

    "http://localhost/api/permissions/assign/user" \
    -H "Content-Type: application/json" \
    -H "Accept: application/json" \
    -d '{"user_id":"officiis","permission_id":"corrupti"}'

```

const url = new URL(
    "http://localhost/api/permissions/assign/user"
);

    "Content-Type": "application/json",
